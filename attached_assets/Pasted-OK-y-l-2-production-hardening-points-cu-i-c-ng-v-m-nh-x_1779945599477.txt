const TTL = 7 * 24 * 60 * 60 * 1000;

function getCenter(): string | null {
  try {
    const raw = localStorage.getItem("edu_center");
    if (!raw) return null;

    const parsed = JSON.parse(raw);
    if (!parsed?.value || !parsed?.ts) return null;

    if (Date.now() - parsed.ts > TTL) {
      localStorage.removeItem("edu_center");
      return null;
    }

    return parsed.value;
  } catch (err) {
    // corrupted storage recovery
    localStorage.removeItem("edu_center");
    return null;
  }
}
