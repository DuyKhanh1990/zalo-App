🔥 FINAL IMPLEMENTATION:
const TTL = 7 * 24 * 60 * 60 * 1000;

function setCenter(center: string) {
  localStorage.setItem("edu_center", JSON.stringify({
    value: center,
    ts: Date.now()
  }));
}

function getCenter() {
  const raw = localStorage.getItem("edu_center");
  if (!raw) return null;

  const { value, ts } = JSON.parse(raw);

  if (Date.now() - ts > TTL) {
    localStorage.removeItem("edu_center");
    return null;
  }

  return value;
}
